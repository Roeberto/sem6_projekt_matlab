\section{Model matematyczny}

\subsection{RГіwnanie bilansu masy}

Przy zaЕӮoЕјeniu staЕӮej gДҷstoЕӣci cieczy, bilans objДҷtoЕӣci w~zbiorniku
o~staЕӮym polu przekroju $A$:
\begin{equation}
A\,\dot h(t) \;=\; Q_{in}(t) - Q_{out}(t) - d(t).
\end{equation}

\subsection{RГіwnanie Torricellego --- odpЕӮyw}

PrДҷdkoЕӣДҮ wypЕӮywu cieczy przez otwГіr na wysokoЕӣci dna wynika z~zasady
zachowania energii (Bernoulli) i~wynosi $v = \sqrt{2gh}$. Z~uwzglДҷdnieniem
wspГіЕӮczynnika kontrakcji strumienia $\mu_v$:
\begin{equation}
Q_{out}(t) \;=\; \mu_v A_v \sqrt{2\,g\,h(t)}.
\end{equation}

\subsection{PeЕӮny model nieliniowy}

Po podstawieniu:
\begin{wzor}[Model nieliniowy]
\begin{equation}
\dot h(t) \;=\; \frac{1}{A}\Big[\,Q_{in}(t) \;-\; \mu_v A_v \sqrt{2 g\,h(t)} \;-\; d(t)\,\Big].
\label{eq:nonlin}
\end{equation}
\end{wzor}

\section{Punkt pracy}

Stan ustalony ($\dot h=0$, $d=0$, $h=h_{ref}$) wymaga rГіwnoЕӣci dopЕӮywu
i~odpЕӮywu:
\begin{equation}
Q_{in,0} \;=\; Q_{out,0} \;=\; \mu_v A_v \sqrt{2\,g\,h_{ref}}.
\end{equation}
Dla przyjДҷtych parametrГіw:
\[
Q_{in,0} = 0{,}97 \cdot \pi\,(0{,}04)^2 \cdot \sqrt{2 \cdot 9{,}81 \cdot 1{,}0}
\approx \SI{0.02160}{m^3/s}.
\]

\section{Linearyzacja wokГіЕӮ punktu pracy}

Wprowadzamy odchyЕӮki: $\Delta h = h - h_{ref}$, $\Delta Q = Q_{in}-Q_{in,0}$.
RozwiniДҷcie prawej strony~\eqref{eq:nonlin} w~szereg Taylora pierwszego rzДҷdu
wokГіЕӮ $(h_{ref},Q_{in,0})$ daje:
\begin{equation}
\Delta\dot h \;=\; a\,\Delta h + b\,\Delta Q + e\,d,
\end{equation}
gdzie
\begin{align}
a &= \left.\frac{\partial f}{\partial h}\right|_{0}
  = \frac{1}{A}\cdot\left(-\mu_v A_v\cdot\frac{2g}{2\sqrt{2g h_{ref}}}\right)
  = -\,\frac{\mu_v A_v}{A}\sqrt{\frac{g}{2\,h_{ref}}}, \\
b &= \left.\frac{\partial f}{\partial Q_{in}}\right|_{0}
  = \frac{1}{A}, \qquad
e = -\frac{1}{A}.
\end{align}

\begin{wzor}[Model liniowy ciД…gЕӮy]
\begin{equation}
\Delta\dot h(t) = a\,\Delta h(t) + b\,\Delta Q(t), \qquad y(t)=\Delta h(t).
\end{equation}
\end{wzor}

WartoЕӣci numeryczne:
\[
a \approx -\SI{3.437e-3}{s^{-1}}, \qquad
b = 1/\pi \approx 0{,}3183, \qquad
T_{ol} = -\tfrac{1}{a} \approx \SI{291}{s}.
\]

\section{Projekt regulatora LQR}

\subsection{WskaЕәnik jakoЕӣci}

Dla ukЕӮadu ciД…gЕӮego SISO poszukujemy sterowania minimalizujД…cego caЕӮkowy
wskaЕәnik jakoЕӣci:
\begin{equation}
J \;=\; \int_{0}^{\infty}\!\!\Big(q\,\Delta h^{2}(t) + r\,\Delta Q^{2}(t)\Big)\,dt,
\qquad q=100,\;\; r=1.
\end{equation}

Waga $q$ karze odchyЕӮki poziomu, $r$ --- wysiЕӮek sterowania. Stosunek
$q/r$ reguluje kompromis szybkoЕӣДҮ--energia.

\subsection{Algebraiczne rГіwnanie Riccatiego (ARE)}

Optymalne sterowanie statyczne $\Delta Q = -K\,\Delta h$ otrzymuje siДҷ
z~rozwiД…zania ARE:
\begin{equation}
A^{T}P + PA - PBR^{-1}B^{T}P + Q = 0.
\end{equation}
Dla skalarnego ukЕӮadu ($A\to a$, $B\to b$, $P\in\mathbb{R}$):
\begin{equation}
2aP - \frac{b^{2}}{r}P^{2} + q = 0.
\end{equation}
RozwiД…zanie dodatnie (wymagane przez stabilnoЕӣДҮ):
\begin{equation}
P = \frac{r}{b^{2}}\left(a + \sqrt{a^{2} + \frac{b^{2}q}{r}}\right).
\end{equation}

\subsection{Wzmocnienie LQR}

\begin{wzor}[Wzmocnienie regulatora]
\begin{equation}
K_{LQR} \;=\; \frac{bP}{r} \;=\; \frac{a + \sqrt{a^{2} + b^{2}q/r}}{b}
\;\approx\; 9{,}989.
\end{equation}
\end{wzor}

\subsection{Biegun zamkniДҷtej pДҷtli}

\begin{equation}
\lambda_{cl} = a - b K_{LQR} \approx -\SI{3.18}{s^{-1}},
\qquad T_{cl} = -\tfrac{1}{\lambda_{cl}} \approx \SI{0.314}{s}.
\end{equation}
UkЕӮad zamkniДҷty jest $\sim$\,925 razy szybszy od obiektu.

\section{Prawo sterowania}

Po powrocie do~zmiennych fizycznych:
\begin{wzor}[Sterowanie pompy --- model ciД…gЕӮy]
\begin{equation}
\boxed{\;Q_{in}(t) \;=\; Q_{in,0} \;-\; K_{LQR}\,\big(h(t)-h_{ref}\big),\;}
\end{equation}
\end{wzor}
z~saturacjД… fizycznД… pompy $0 \le Q_{in}(t) \le Q_{pump}$.

\section{Wnioski}

\begin{itemize}\setlength{\itemsep}{2pt}
\item Obiekt jest obiektem pierwszego rzДҷdu nieliniowym ---
      linearyzacja w~punkcie pracy jest wystarczajД…ca, bo LQR utrzymuje
      $h$ blisko $h_{ref}$, gdzie bЕӮД…d linearyzacji jest maЕӮy.

      co odpowiada czasowi regulacji rzДҷdu sekund.
\item Saturacja $[0,Q_{pump}]$ zabezpiecza przed ujemnym przepЕӮywem
      (pompa nie wypompowuje wstecznie) oraz przed przekroczeniem
      wydajnoЕӣci fizycznej pompy.
\item W~wersji ciД…gЕӮej nie uЕјywamy histerezy ani przeЕӮД…czania ON/OFF
      obecnych w~oryginalnym skrypcie \texttt{sem6\_projekt\_v1.m};
      pompa pracuje z~ciД…gle modulowanym wydatkiem.
\end{itemize}
